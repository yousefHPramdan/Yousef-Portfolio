import { Toolkit } from '@youpify/core';

// Initialize with optimized settings
const app = new Toolkit({
  mode: 'production',
  hydration: true
});

app.on('cart:add', (item) => {
  console.log(`Added: ${item.name}`);
  app.trigger('ui:toast', {
    message: 'Success!',
    type: 'jade'
  });
});
